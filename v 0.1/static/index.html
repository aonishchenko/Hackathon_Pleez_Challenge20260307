# Analyse a restaurant file via curl curl -X POST http://localhost:8000/api/analyse \ -F 'file=@restaurant_data.xlsx' # Response (truncated) { "success": true, "report": { "restaurant_id": "R29", "cuisine": "Burgers", "predicted_wins": 3, "avg_predicted_roi": 52.4, "top_actions": [ { "rank": 1, "promo_type": "free-delivery", "predicted_roi": 70.8, "is_predicted_win": true, "implementation": "Activate Free Delivery Thu–Sun..." } // ... 4 more actions ], "hidden_truths": [ /* context-specific insights */ ] } }
# Python example import requests with open("restaurant_data.xlsx", "rb") as f: response = requests.post( "http://localhost:8000/api/analyse", files={"file": ("data.xlsx", f, "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet")} ) report = response.json()["report"] for action in report["top_actions"]: print(f"#{action['rank']} {action['label']}: {action['predicted_roi']}% ROI")
